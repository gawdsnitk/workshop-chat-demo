    Basic syntax & Selectors

    selector {
        style-attribute: value;
        .
        .
    }
    /*A practical example*/
    body {
        background-color: red;
        padding: 10px;
        margin: 10px;
    }
    /*The universal selector * */
    * {
        box-sizing: border-box;
        padding: 10px;
        margin: 10px;
    }
    /*The id selector*/
    #id {
        height: 30px;
        width: 40px;
        border: 1px solid #000000;
    }
    /*The class selector*/
    .classname {
        height: 30px;
        width: 40px;
        border: 1px solid #000000;
    }
